#version 400 core

in vec2 _position;

out vec4 _clipSpace;

uniform mat4 _projectionMatrix;
uniform mat4 _viewMatrix;
uniform mat4 _modelMatrix;

void main(void) {
	 _clipSpace = _projectionMatrix * _viewMatrix * _modelMatrix * vec4(_position.x, 0.0, _position.y, 1.0);
	 gl_Position = _clipSpace;
}
